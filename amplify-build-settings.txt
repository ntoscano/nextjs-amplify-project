When updating your Pulumi configuration, replace the existing buildSpec with this one:

buildSpec: `
version: 1
frontend:
  phases:
    preBuild:
      commands:
        - npm ci
        - echo "Adding explicit NODE_OPTIONS to avoid SSR issues"
        - export NODE_OPTIONS="--max-old-space-size=4096"
    build:
      commands:
        - echo "Building Next.js app"
        - npm run build
        - echo "Next.js build completed"
  artifacts:
    baseDirectory: .next
    files:
      - '**/*'
  cache:
    paths:
      - node_modules/**/*
`,

This avoids potential issues with the automated SSR build process in Amplify that might be adding the target property.
